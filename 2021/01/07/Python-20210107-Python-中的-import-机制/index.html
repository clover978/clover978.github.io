Python 中的 import 机制

Posted on 2021-01-07 In Python

  • module import 

      import m
  from m import *
  
  # empty

  • package import

      import m
  from m import *

  import C
  from C import *
  
  import C
  from C import *
  
      # empty

  • relative import

          import m
      from m import *

      import C
      from C import *
      
      import C
      from C import *
      
      # import A.m
      from m import *

      # import A.C
      from C import *
      
      # import A.C
      from C import *
      
      # import A.m
      from .m import *

      # import A.C
      from .C import *
      
      # import A.C
      from .C import *
      
      from ..A import *
      
      # import A.m
      from .m import *

      # import A.C
      from .C import *
      
      # import A.C
      from .C import *
      
      from ..A import *
